                            //get the values from interface
                            function getValues() {
                                //get values from page
                                let startValue = document.getElementById("startValue").value;
                                let endValue = document.getElementById("endValue").value;
                            
                                //parse into integers
                                startValue = parseInt(startValue);
                                endValue = parseInt(endValue);
                                //alert("The start value: " + startValue);
                                
                                if (Number.isInteger(startValue) && Number.isInteger(endValue)) {
                                    let numbers = generateNumbers(startValue, endValue);
                                    //call display numbers
                                    displayNumbers(numbers);       
                            
                                } else {
                                    alert("You must enter a valid number");
                                }
                              
                                
                            
                            }
                            
                            //generate numbers to start value to end value
                            function generateNumbers(startValue, endValue) {
                                let numbers = [];
                            
                                for (let index = startValue; index <= endValue; index++) {
                                   numbers.push(index);
                                }
                            
                                return numbers;
                            }
                            
                            //Display the numbers and mark even number in bold
                            function displayNumbers(numbers) {
                                let templateRows = ""
                                for (let index = 0; index < numbers.length; index++) {
                                    let number = numbers[index];
                                    let className = "even";
                            
                                    if (number % 2 == 0) {
                                        className == "even";
                                    } else {
                                        className = "odd";
                                    }
                                    //this render correctly with prism 

                                    templateRows += `${number}`; 
                            
                                }
                                document.getElementById("results").innerHTML = templateRows;
                            }

The Code is structed in three functions

Function getValues

The function getValues() will assign to 2 variables, startValue and endValue, the values captured from the html inputs with ids, startValue and endValue, respectively, converting these values into integers using the function parseInt(). If the content of these 2 variables really was converted to integers, the code will then call the defined functions generateNumbers() and displayNumbers(). The return of the function "generateNumbers", which is an array, will be assigned to a variable called numbers, which will then be passed as argument to the function displayNumbers() to be displayed. Otherwise, it will emit an alert, asking the user to enter a valid number.


Function generateNumbers

The function generateNumbers() will declare an arrray called numbers, initializing it trough an iteration process, each of its positions. This function contains 2 parameters, startValue and endValue, which will determine where the iteration process will start, and where it will end. After the iteration process is complete, the function will return the array of numbers back to the caller.


Function displayNumbers

The function displayNumbers() will process an array of numbers using a for loop. Each item of this array will be assigned to a variable called number, at each iteration, analyzing if the value of this item is an even or odd number. The templateRows variable will be initialized at each iteration with an html instruction, which contains javascript elements informing the value of each position of the array of numbers, and if each of its values are even or odd, rendering this information at the html document.
